Control Structures

// if-else
if (age >= 18) {
    System.out.println("Adult");
} else {
    System.out.println("Minor");
}

// for loop
for(int i = 0; i < 5; i++) {
    System.out.println(i);
}

// Enhanced for (arrays/collections)
for(String name : names) {
    System.out.println(name);
}

// while loop
while (age < 21) {
    System.out.println("Not eligible to vote");
    age++;
}

// switch case
int day = 1;
switch(day) {
    case 1: System.out.println("Monday"); break;
    case 2: System.out.println("Tuesday"); break;
    default: System.out.println("Unknown");
}

// break/continue
for(int i = 0; i < 10; i++) {
    if (i == 5) continue; // Skip i=5
    if (i == 7) break;    // Exit loop
}

OOP Concepts

Classes & Objects

public class Student {
    private String name; // Encapsulation
    private int age;

    public Student(String name, int age) { // Constructor
        this.name = name;
        this.age = age;
    }

    public String getName() { return name; } // Getter
    public void setName(String name) { this.name = name; } // Setter
}

// Usage
Student alice = new Student("Alice", 22);

Inheritance

class Vehicle {
    void start() { System.out.println("Engine starting"); }
}
class Car extends Vehicle {
    void accelerate() { System.out.println("Accelerating"); }
}

Polymorphism

Vehicle car = new Car(); // Upcasting
car.start(); // Outputs "Engine starting"

Interfaces & Abstract Classes

interface Driveable {

abstract class ElectricVehicle {

Best Practices

Common Errors & Fixes

NullPointerException

String name = null;
System.out.println(name.length()); // Causes NPE
// Fix: Check for null before using
if (name != null) { ... }

ArrayIndexOutOfBoundsException

System.out.println(marks[5]); // Index 5 out of bounds for length 3
// Fix: Check array bounds
if (index >= 0 && index < marks.length) { ... }
